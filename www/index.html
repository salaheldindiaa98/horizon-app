  3. Paste this:
package com.salah.horizon;

import android.appwidget.AppWidgetManager;
import android.appwidget.AppWidgetProvider;
import android.content.Context;
import android.content.SharedPreferences;
import android.widget.RemoteViews;
import org.json.JSONObject;
import java.text.SimpleDateFormat;
import java.util.Date;
import java.util.Locale;

public class HorizonWidget extends AppWidgetProvider {

    @Override
    public void onUpdate(Context context, AppWidgetManager mgr, int[] ids) {
        for (int id : ids) updateWidget(context, mgr, id);
    }

    static void updateWidget(Context context, AppWidgetManager mgr, int id) {
        RemoteViews views = new RemoteViews(context.getPackageName(), R.layout.widget_horizon);
        try {
            // Try both SharedPreference store names Capacitor may use
            SharedPreferences prefs = context.getSharedPreferences("CAPStorage", Context.MODE_PRIVATE);
            String json = prefs.getString("featured_event", null);
            if (json == null) {
                prefs = context.getSharedPreferences("CapacitorStorage", Context.MODE_PRIVATE);
                json = prefs.getString("featured_event", null);
            }
            if (json != null) {
                JSONObject ev = new JSONObject(json);
                String name = ev.getString("name");
                long targetMs = ev.getLong("targetMs");
                long diff = targetMs - System.currentTimeMillis();
                views.setTextViewText(R.id.widget_event_name, name);
                if (diff > 0) {
                    long d = diff / 86400000L;
                    long h = (diff % 86400000L) / 3600000L;
                    long m = (diff % 3600000L) / 60000L;
                    views.setTextViewText(R.id.widget_countdown, d + "d  " + h + "h  " + m + "m");
                    String date = new SimpleDateFormat("MMM d, yyyy", Locale.getDefault())
                            .format(new Date(targetMs));
                    views.setTextViewText(R.id.widget_date, date);
                } else {
                    views.setTextViewText(R.id.widget_countdown, "Has passed");
                    views.setTextViewText(R.id.widget_date, "");
                }
            } else {
                views.setTextViewText(R.id.widget_event_name, "Open Horizon to load");
                views.setTextViewText(R.id.widget_countdown, "");
                views.setTextViewText(R.id.widget_date, "");
            }
        } catch (Exception e) {
            views.setTextViewText(R.id.widget_event_name, "Horizon");
            views.setTextViewText(R.id.widget_countdown, "Tap to open app");
            views.setTextViewText(R.id.widget_date, "");
        }
        mgr.updateAppWidget(id, views);
    }
}

Step 1.5 — Create the GitHub Actions workflow

This is the file that tells GitHub's cloud computers how to build your APK automatically every time you make a change.

  1. Tap Add file → Create new file
  2. Filename: .github/workflows/build.yml (the dot at the start is required)
  3. Paste this:
name: Build Horizon APK

on:
  push:
    branches: [main]
  workflow_dispatch:

jobs:
  build:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'

      - name: Install npm packages
        run: npm ci

      - name: Add Android platform
        run: npx cap add android

      - name: Copy widget layout
        run: |
          mkdir -p android/app/src/main/res/layout
          mkdir -p android/app/src/main/res/xml
          cp widget_files/widget_horizon.xml android/app/src/main/res/layout/widget_horizon.xml
          cp widget_files/widget_info.xml android/app/src/main/res/xml/widget_info.xml

      - name: Copy widget Java class
        run: |
          mkdir -p android/app/src/main/java/com/salah/horizon
          cp widget_files/HorizonWidget.java android/app/src/main/java/com/salah/horizon/HorizonWidget.java

      - name: Add widget and permissions to AndroidManifest
        run: |
          python3 << 'PYEOF'
          with open('android/app/src/main/AndroidManifest.xml', 'r') as f:
              content = f.read()

          # Add permissions if not already present
          permissions = '''    <uses-permission android:name="android.permission.POST_NOTIFICATIONS"/>
      <uses-permission android:name="android.permission.SCHEDULE_EXACT_ALARM"/>
      <uses-permission android:name="android.permission.USE_EXACT_ALARM"/>
      <uses-permission android:name="android.permission.RECEIVE_BOOT_COMPLETED"/>'''

          if 'POST_NOTIFICATIONS' not in content:
              content = content.replace('<application', permissions + '\n\n    <application', 1)

          # Add widget receiver
          widget_receiver = '''
          <receiver
              android:name=".HorizonWidget"
              android:exported="true">
              <intent-filter>
                  <action android:name="android.appwidget.action.APPWIDGET_UPDATE" />
              </intent-filter>
              <meta-data
                  android:name="android.appwidget.provider"
                  android:resource="@xml/widget_info" />
          </receiver>'''

          if 'HorizonWidget' not in content:
              content = content.replace('</application>', widget_receiver + '\n    </application>', 1)

          with open('android/app/src/main/AndroidManifest.xml', 'w') as f:
              f.write(content)

          print("AndroidManifest.xml updated successfully")
          PYEOF

      - name: Sync web assets into Android
        run: npx cap sync android

      - name: Set up Java 17
        uses: actions/setup-java@v4
        with:
          java-version: '17'
          distribution: 'temurin'

      - name: Set up Android SDK
        uses: android-actions/setup-android@v3

      - name: Accept Android SDK licenses
        run: yes | sdkmanager --licenses || true

      - name: Build debug APK
        working-directory: android
        run: ./gradlew assembleDebug --no-daemon

      - name: Upload APK
        uses: actions/upload-artifact@v4
        with:
          name: Horizon-APK
          path: android/app/build/outputs/apk/debug/app-debug.apk
          retention-days: 30

PHASE 4 — Update Horizon to Support Notifications and Widget

(10 minutes)

The Horizon app you built with Claude stores reminder times but doesn't yet fire real notifications. You need to add a small piece of code to www/index.html that talks to Android's notification system via Capacitor.

Step 4.1 — Open index.html in the nano editor

Step 4.2 — Find the saveData function

Press Ctrl + W, type function saveData, press Enter.

You should land on this line:

function saveData(){ localStorage.setItem(STORE_KEY, JSON.stringify(data)); }

Step 4.3 — Replace it with the Capacitor-aware version

Delete that line and replace it with:

function saveData(){
  localStorage.setItem(STORE_KEY, JSON.stringify(data));
  updateWidgetData();
  rescheduleAllNotifications();
}

async function updateWidgetData(){
  if(!window.Capacitor?.isNativePlatform()) return;
  try {
    const { Preferences } = window.Capacitor.Plugins;
    const all = getOrderedEvents();
    const featured = all.find(e=>e.featured) || all[0];
    if(!featured) return;
    const target = combinedDate(featured);
    await Preferences.set({
      key: 'featured_event',
      value: JSON.stringify({ name: featured.name, targetMs: target.getTime() })
    });
  } catch(e){ console.warn('Widget update failed:', e); }
}

async function rescheduleAllNotifications(){
  if(!window.Capacitor?.isNativePlatform()) return;
  try {
    const { LocalNotifications } = window.Capacitor.Plugins;
    const perm = await LocalNotifications.requestPermissions();
    if(perm.display !== 'granted') return;
    // Cancel all pending notifications then reschedule
    const pending = await LocalNotifications.getPending();
    if(pending.notifications.length > 0){
      await LocalNotifications.cancel({ notifications: pending.notifications });
    }
    const toSchedule = [];
    const now = new Date();
    data.events.forEach(ev => {
      (ev.reminders || []).forEach((minsBefore, idx) => {
        const target = combinedDate(ev);
        const fireAt = new Date(target.getTime() - minsBefore * 60000);
        if(fireAt > now){
          const label = minsBefore === 0 ? 'is now!' :
            minsBefore < 60 ? `in ${minsBefore} minutes` :
            minsBefore < 1440 ? `in ${Math.round(minsBefore/60)} hours` :
            `tomorrow`;
          toSchedule.push({
            title: '⏱ ' + ev.name,
            body: ev.name + ' ' + label,
            id: Math.abs(simpleHash(ev.id + '_' + idx)) % 2000000000,
            schedule: { at: fireAt, allowWhileIdle: true },
            sound: 'default',
            channelId: 'horizon_reminders'
          });
        }
      });
    });
    if(toSchedule.length > 0){
      await LocalNotifications.schedule({ notifications: toSchedule });
    }
  } catch(e){ console.warn('Notification scheduling failed:', e); }
}

function simpleHash(str){
  let h = 0;
  for(let i = 0; i < str.length; i++){ h = Math.imul(31, h) + str.charCodeAt(i) | 0; }
  return h;
}

TROUBLESHOOTING

Build failed — "Gradle build failed"

This is the most common error. Causes and fixes:

Fix 1: Tap Re-run all jobs in GitHub Actions — sometimes it's a temporary network error

Fix 2: Check that your package.json is exactly as written in Step 1.4 — even one missing comma breaks it

Fix 3: Check the exact error message in the failed step — tap the step to expand it. Look for the last line before the red error.

"npx cap add android" failed in the workflow

This means the package.json has an error or the npm install step failed. Check that package.json has no typos.

Widget shows "Open Horizon to load" even after opening the app

This means the Capacitor Preferences plugin isn't writing to the location the widget expects. Try this:

  1. Open Horizon → create or tap an event (this triggers saveData)
  2. Wait 30 minutes for the widget to refresh
  3. If still not working, go to your repo → widget_files/HorizonWidget.java → find the line "CAPStorage" and change it to "CapacitorStorage" → commit → rebuild → reinstall

Notifications are not firing

  1. Go to Settings → Apps → Horizon → Notifications → make sure all notification categories are On
  2. Go to Settings → Battery → Horizon → set to Unrestricted (very important on Android 14 — restricted mode kills background notifications)
  3. Go to Settings → Apps → Special app access → Alarms & reminders → Horizon → turn On

"Install blocked" when installing the APK

  1. Go to Settings → Security → Install unknown apps
  2. Find your file manager or browser → enable Allow from this source

Termux says "command not found" for node or git

Re-run the install commands:

pkg install -y nodejs-lts git

I made a mistake in nano and want to start over

Press Ctrl + X without saving (press N when asked to save). Then run:

git checkout www/index.html

This restores the original file from GitHub.
